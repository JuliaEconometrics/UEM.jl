using DataFrames, RDatasets, UEM
df = dataset("plm", "Crime")
pool!(df, [:Region, :SMSA])
fm = @formula(CRMRTE ~ PrbConv + PrBarr + Region)
iv = @formula(Density + WSer ~ PctYMle + WFed)
  • Pooling OLS model with OLS Variance-Covariance Estimator
model = uem(:PO, fm, df)
coeftable(model)
